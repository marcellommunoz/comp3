void main { 
	newVar int arg1 ;
	newVar int arg2 ;
	newVar int contador;
	contador = 1;
	while(arg2 > 0){
		troca( arg1, arg2, contador );
		contador = contador - 1;
	}
	System.output ( arg1 ) ;
}

func bool troca ( int arg1 , int arg2, int diff ) {
	arg1 = arg1 + diff;
	arg2 = arg2 - diff;
	if(arg2 > 0) {
		return true;
	}
	return false
}
